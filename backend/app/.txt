from fastapi import FastAPI, APIRouter
from sqlalchemy import text
from sqlalchemy.exc import ProgrammingError
from app.core.db import async_session, engine
from app.models.user import Base
from app.api.auth import router as auth_router
import uvicorn

app = FastAPI()

router = APIRouter()

@app.on_event("startup")
async def startup_event():
    try:
        # Check if the table exists and create it if it doesn't
        async with engine.begin() as conn:
            result = await conn.run_sync(Base.metadata.create_all)
            print("\033[92mTables created successfully (if not already present)\033[0m")
        
        # Test database connection
        async with async_session() as session:
            await session.execute(text("SELECT 1"))
        print("\033[92mDatabase connection successful\033[0m")
    except ProgrammingError as e:
        print(f"\033[91mError creating tables or connecting to the database: {e}\033[0m")
    except Exception as e:
        print(f"\033[91mUnexpected error: {e}\033[0m")

# Include the auth router
app.include_router(auth_router, prefix="/auth")

@app.get("/")
def read_root():
    return {"Hello": "World"}

if __name__ == "__main__":
    uvicorn.run(app, host="0.0.0.0", port=8000)
